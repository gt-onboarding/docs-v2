ソーシャルログイン操作を行うためのオプション

```ts Example
export interface FederatedLoginOptions {
  /** 使用するソーシャル接続の名前 */
  connection: string;
  /** その他の追加カスタムオプション */
  [key: string]: string | number | boolean;
}
```
