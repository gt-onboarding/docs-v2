```ts Example
export interface ScreenMembersOnResetPasswordMfaWebAuthnPlatformChallenge extends ScreenMembers {
  /**
   * WebAuthnに必要な公開鍵クレデンシャルリクエストオプション（チャレンジ）を取得するための便利なゲッター。
   * `screen.data.passkey.public_key`から派生します。
   * `screen.data.passkey`または`screen.data.passkey.public_key`が利用できない場合は`null`になります。
   * @type {PasskeyRead['public_key'] | null}
   */
  publicKey: PasskeyRead['public_key'] | null;

  /**
   * 「このデバイスを記憶する」オプションを表示するかどうかを取得するための便利なゲッター。
   * `screen.data.show_remember_device`から派生します。指定されていない場合のデフォルトは`false`です。
   * @type {boolean}
   */
  showRememberDevice: boolean;
}
```
