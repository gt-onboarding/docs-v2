```ts Example
export interface PhonePrefix {
  /** 国名 */
  country: string;
  /** 国コード (例: 'US', 'GB') */
  country_code: string;
  /** 電話番号プレフィックス (例: '+1', '+44') */
  phone_prefix: string;
}
```
