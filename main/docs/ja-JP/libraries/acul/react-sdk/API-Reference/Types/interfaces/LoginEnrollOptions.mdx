title: "LoginEnrollOptions"

選択した MFA（多要素認証）要素で認証を続行するためのオプション

```ts Example
export interface LoginEnrollOptions {
  /** ログインに使用する要素を示すアクション */
  action: MfaLoginFactorType;
  /** その他のカスタムオプション */
  [key: string]: string | number | boolean | undefined;
}
```
