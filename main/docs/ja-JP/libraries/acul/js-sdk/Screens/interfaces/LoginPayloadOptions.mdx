---
title: "LoginPayloadOptions"
---

ログイン処理を行うためのオプション

```ts Example
export interface LoginOptions {
  /** ログインに使用するユーザー名/メールアドレス */
  username: string;
  /** 認証用パスワード */
  password: string;
  /** 必要な場合に指定するオプションのCAPTCHA値 */
  captcha?: string;
  /** 追加のカスタムオプション */
  [key: string]: string | number | boolean | undefined;
}
```


<div id="indexable">
  ## インデックス可能
</div>

\[`key`: `string`\]: `string` \| `number` \| `boolean` \| `undefined`

任意の追加のカスタムオプション
