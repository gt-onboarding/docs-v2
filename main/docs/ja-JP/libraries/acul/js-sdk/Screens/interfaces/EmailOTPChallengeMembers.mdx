---
title: "EmailOTPChallengeMembers"
---

Email による OTP チャレンジ画面のインターフェースを表します。

```ts Example
export interface EmailOTPChallengeMembers extends BaseMembers {
  screen: ScreenMembersOnEmailOTPChallenge;
  /**
   * ユーザーが入力したOTPコードを送信します。
   * @param options 送信に含めるオプションパラメータ。
   */
  submitCode(options: OtpCodeOptions): Promise<void>;
  /**
   * ユーザーのメールアドレスに新しいOTPコードの送信をリクエストします。
   * @param options 再送信リクエストに含めるオプションパラメータ。
   */
  resendCode(options?: CustomOptions): Promise<void>;
  /**
   * この画面のタイムアウト管理機能を備えた再送信機能を取得します
   * @param options 再送信機能の設定オプション
   */
  resendManager(options?: StartResendOptions): ResendControl;
}
```
