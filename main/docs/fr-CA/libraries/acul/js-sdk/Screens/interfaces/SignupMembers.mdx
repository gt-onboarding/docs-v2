title: "SignupMembers"

````ts Example
export interface SignupMembers extends BaseMembers {
  screen: ScreenMembersOnSignup;
  transaction: TransactionMembersOnSignup;
  signup(payload: SignupOptions): Promise<void>;
  federatedSignup(payload: FederatedSignupOptions): Promise<void>;
  pickCountryCode(payload?: CustomOptions): Promise<void>;
  validatePassword(password: string): PasswordValidationResult;
  /**
 * Retourne une liste des identifiants activés (p. ex., courriel, téléphone, nom d'utilisateur)
 * en fonction de l'état actuel de la transaction.
 *
 * Les identifiants peuvent être obligatoires ou facultatifs selon la stratégie de connexion
 * et la configuration fournie durant le flux d'authentification ou d'inscription.
 *
 * @returns Un tableau d'objets {@link Identifier} activés, ou `null` si la transaction n'est pas initialisée.
 *
 * @example
 * ```ts
 * const identifiers = authClient.getSignupIdentifiers();
 * if (identifiers) {
 *   identifiers.forEach(({ type, required }) => {
 *     console.log(`${type} is ${required ? 'required' : 'optional'}`);
 *   });
 * }
 * ```
 * @utilityFeature
 * @see Identifier
 */
  getSignupIdentifiers(): Identifier[] | null;
  validateUsername(username: string): UsernameValidationResult;
}
````

<ParamField body="getSignupIdentifiers" type={<span><a href="/docs/libraries/acul/js-sdk/Screens/interfaces/Identifier">Identifier</a>[]</span>}>
  **`Utility Feature`**

  Renvoie une liste des identifiants activés (par exemple, courriel, téléphone, nom d&#39;utilisateur)
  en fonction du state de la transaction en cours.

  Les identifiants peuvent être obligatoires ou facultatifs selon la stratégie de connexion
  et la configuration fournie au cours du flux d’authentification ou d’inscription.

  Renvoie un tableau d’objets [Identifier](/docs/fr-CA/libraries/acul/js-sdk/Screens/interfaces/Identifier) activés, ou `null` si la transaction n’est pas initialisée.

  ```ts Example
  const identifiers = authClient.getSignupIdentifiers();
  if (identifiers) {
    identifiers.forEach(({ type, required }) => {
      console.log(`${type} is ${required ? 'required' : 'optional'}`);
    });
  }
  ```
</ParamField>
