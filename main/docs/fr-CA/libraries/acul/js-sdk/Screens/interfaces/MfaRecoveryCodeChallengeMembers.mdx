---
title: "MfaRecoveryCodeChallengeMembers"
---

Interface décrivant les membres de l’écran de défi par code de récupération de l’AMF (MFA).

```ts Example
export interface MfaRecoveryCodeChallengeMembers extends BaseMembers {
  client: ClientMembers;
  organization: OrganizationMembers;
  prompt: PromptMembers;
  screen: ScreenMembers;
  transaction: TransactionMembers;

  /**
   * Poursuit le processus avec le code de récupération fourni.
   * @param payload - Les options de continuation contenant le code de récupération et les options personnalisées facultatives.
   * @returns Une promesse qui se résout lorsque la continuation est réussie.
   */
  continue(payload: ContinueOptions): Promise<void>;

  /**
   * Navigue vers l'écran où l'utilisateur peut choisir une autre méthode AMF (MFA).
   * @param payload Charge utile facultative.
   * @returns Une promesse qui se résout lorsque la navigation est terminée.
   */
  tryAnotherMethod(payload?: CustomOptions): Promise<void>;
}
```
