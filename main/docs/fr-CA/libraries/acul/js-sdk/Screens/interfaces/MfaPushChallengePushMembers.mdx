Interface qui définit les méthodes et propriétés disponibles pour l&#39;écran `mfa-push-challenge-push`

```ts Example
export interface MfaPushChallengePushMembers extends BaseMembers {
  screen: ScreenMembersOnMfaPushChallengePush;
  untrustedData: UntrustedDataMembersOnMfaPushChallengePush;

  /**
   * Poursuit le défi de notification push
   * @param payload Options personnalisées facultatives à inclure avec la requête
   */
  continue(payload?: WithRememberOptions): Promise<void>;

  /**
   * Renvoie la notification push
   * @param payload Options personnalisées facultatives à inclure avec la requête
   */
  resendPushNotification(payload?: WithRememberOptions): Promise<void>;

  /**
   * Bascule vers la saisie manuelle du code de vérification
   * @param payload Options personnalisées facultatives à inclure avec la requête
   */
  enterCodeManually(payload?: CustomOptions): Promise<void>;

  /**
   * Permet d'essayer une autre méthode d'authentification
   * @param payload Options personnalisées facultatives à inclure avec la requête
   */
  tryAnotherMethod(payload?: CustomOptions): Promise<void>;

  /** 
   * Gère l'interrogation de l'état du défi push AMF (MFA)
   * @param options Options de configuration pour l'interrogation de type {@link MfaPollingOptions}
   * @returns Un objet pour contrôler le processus d'interrogation
  */
  pollingManager(options: MfaPollingOptions): MfaPushPollingControl;
}
```
