---
title: "OrganizationPickerMembers"
---

Interface qui définit les méthodes et propriétés disponibles pour l&#39;écran de sélection d&#39;organisation

```ts Example
export interface OrganizationPickerMembers extends BaseMembers {
  /**
   * Sélectionne une organisation pour l'utilisateur.
   * @param payload Les options contenant l'identifiant de l'organisation.
   */
  selectOrganization(payload: { organization: string; state: string }): Promise<void>;

  /**
   * Ignore la sélection de l'organisation et poursuit avec le compte personnel de l'utilisateur.
   * @param payload Options personnalisées facultatives à inclure avec la requête.
   */
  skipOrganizationSelection(payload?: CustomOptions): Promise<void>;
}
```
