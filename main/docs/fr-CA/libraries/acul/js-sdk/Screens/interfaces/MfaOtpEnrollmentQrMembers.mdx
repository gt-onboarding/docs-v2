```ts Example
export interface MfaOtpEnrollmentQrMembers extends BaseMembers {
  screen: ScreenMembersOnMfaOtpEnrollmentQr;

  /**
   * Bascule l'affichage.
   * @param payload Options personnalisées facultatives à inclure dans la requête
   */
  toggleView(payload?: CustomOptions): Promise<void>;

  /**
   * Poursuit avec l'action par défaut.
   * @param payload Charge utile contenant le code et les options personnalisées facultatives

  /**
   * Permet d'essayer une autre méthode d'authentification
   * @param payload Options personnalisées facultatives à inclure dans la requête
   */
  tryAnotherMethod(payload?: CustomOptions): Promise<void>;
}
```
