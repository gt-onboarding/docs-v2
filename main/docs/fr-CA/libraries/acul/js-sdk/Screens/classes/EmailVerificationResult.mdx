EmailVerificationResult

```typescript Example
import EmailVerificationResult from '@auth0/auth0-acul-js/email-verification-result';

const emailVerificationResultScreen = new EmailVerificationResult();

// Accéder aux données de l'écran
const status = emailVerificationResultScreen.screen.data?.status;
const loginLink = emailVerificationResultScreen.screen.loginLink;

console.log(`Statut de vérification : ${status}`);
if (loginLink) {
  console.log(`Procéder à la connexion : ${loginLink}`);
  // Dans une interface utilisateur, vous utiliseriez ce lien pour un bouton ou une balise d'ancrage
  // p. ex., <a href={loginLink}>Aller à la connexion</a>
}
```
