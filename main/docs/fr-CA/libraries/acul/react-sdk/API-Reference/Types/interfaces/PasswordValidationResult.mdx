---
title: "Résultat de la validation du mot de passe"
---

```ts Example
export interface PasswordValidationResult {
  /**
   * Overall outcome of all password policy checks.
   *
   * `true` only if every rule in {@link PasswordComplexityRule} has `status` equal to `"valid"`.
   */
  isValid: boolean;

  /**
   * Tableau des règles de mot de passe évaluées.
   *
   * Chaque élément est un {@link PasswordComplexityRule} dont le
   * champ {@link PasswordComplexityRule.status | status} reflète le
   * résultat de validation calculé pour cette règle.
   */
  results: PasswordComplexityRule[];
}
```

<div id="properties">
  ## Propriétés
</div>

<ParamField body="isValid" type="boolean">
  Résultat global de toutes les vérifications de la stratégie de mots de passe.

  `true` uniquement si chaque règle dans [PasswordComplexityRule](/docs/fr-CA/libraries/acul/react-sdk/API-Reference/Types/interfaces/PasswordComplexityRule) a un `status` égal à `"valid"`.
</ParamField>

<ParamField body="results" type={<span><a href="/docs/libraries/acul/react-sdk/API-Reference/Types/interfaces/PasswordComplexityRule">PasswordComplexityRule</a></span>}>
  Tableau des règles de mots de passe évaluées.

  Chaque élément est une [PasswordComplexityRule](/docs/fr-CA/libraries/acul/react-sdk/API-Reference/Types/interfaces/PasswordComplexityRule) dont le champ
  [status](/docs/fr-CA/libraries/acul/react-sdk/API-Reference/Types/interfaces/PasswordComplexityRule#status) reflète le résultat de validation calculé pour cette règle.
</ParamField>
