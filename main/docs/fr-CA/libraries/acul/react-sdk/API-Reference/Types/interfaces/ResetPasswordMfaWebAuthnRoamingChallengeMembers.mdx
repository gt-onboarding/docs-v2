export interface ResetPasswordMfaWebAuthnRoamingChallengeMembers extends BaseMembers {

  /**
 * Accès aux propriétés et données spécifiques de l'écran `reset-password-mfa-webauthn-roaming-challenge`,
 * incluant les options de défi `publicKey` WebAuthn et l'indicateur `showRememberDevice`.
 * @type {ScreenMembersOnResetPasswordMfaWebAuthnRoamingChallenge}
 */
  screen: ScreenMembersOnResetPasswordMfaWebAuthnRoamingChallenge;

  /**
   * Initie le processus d'assertion WebAuthn (défi de clé de sécurité).
   * Cette méthode appellera en interne `navigator.credentials.get()` en utilisant le défi
   * fourni dans `screen.publicKey`.
   * En cas d'interaction réussie avec la clé de sécurité, elle soumet le
   * `PublicKeyCredential` résultant à Auth0 avec `action: "default"`.
   *
   * @param {UseSecurityKeyOptions} [options] - Optionnel. Paramètres pour l'opération,
   * tels que `rememberDevice` (si `screen.showRememberDevice` est true) et d'autres options personnalisées.
   * Le champ `response` (l'identifiant WebAuthn) est géré en interne par le SDK.
   * @returns {Promise<void>} Une promesse qui se résout lorsque la tentative de vérification est soumise.
   * Une opération réussie entraîne généralement une redirection par Auth0.
   * @throws {Error} Lance une erreur si `screen.publicKey` est manquant, si `navigator.credentials.get()`
   * échoue (p. ex., annulation par l'utilisateur, aucun authentificateur trouvé), ou si la soumission du formulaire à Auth0 échoue.
   * Il est recommandé d'intercepter ces erreurs et d'utiliser potentiellement `showError()` pour signaler les problèmes spécifiques à l'API WebAuthn.
   */
  useSecurityKey(options?: UseSecurityKeyOptions): Promise<void>;

  /**
   * Signale une erreur d'API WebAuthn côté client (provenant de `navigator.credentials.get()`) à Auth0.
   * Cette méthode doit être utilisée lorsque `useSecurityKey()` (ou un appel manuel à `navigator.credentials.get()`)
   * échoue en raison d'un problème WebAuthn côté navigateur (p. ex., `NotAllowedError` si l'utilisateur annule l'invite,
   * `NotFoundError` si aucun authentificateur correspondant n'est trouvé, ou un délai d'expiration).
   * Elle soumet les détails de l'erreur avec `action: "showError::{errorDetailsJsonString}"` et une `response` vide.
   *
   * @param {ShowErrorOptions} options - Contient l'objet `error` (avec `name` et `message`
   * provenant de la DOMException de l'API WebAuthn), un indicateur `rememberDevice` optionnel, et toute autre option personnalisée.
   * @returns {Promise<void>} Une promesse qui se résout lorsque le rapport d'erreur est soumis.
   * Auth0 peut réafficher la page avec des informations d'erreur ou rediriger.
   * @throws {Error} Lance une erreur si la soumission du formulaire échoue (p. ex., erreur réseau, état invalide).
   */
  showError(options: ShowErrorOptions): Promise<void>;

  /**
   * Permet à l'utilisateur de se retirer du défi WebAuthn itinérant et de sélectionner une méthode AMF différente.
   * Cette action soumet `action: "pick-authenticator"` à Auth0, ce qui devrait diriger
   * l'utilisateur vers un écran de sélection de facteur AMF.
   *
   * @param {TryAnotherMethodOptions} [options] - Optionnel. Paramètres pour l'opération,
   * tels que `rememberDevice` (si `screen.showRememberDevice` est true) et d'autres options personnalisées.
   * @returns {Promise<void>} Une promesse qui se résout lorsque l'action 'pick-authenticator' est soumise.
   * @throws {Error} Lance une erreur si la soumission du formulaire échoue.
   */
  tryAnotherMethod(options?: TryAnotherMethodOptions): Promise<void>;
}

<ParamField body="showError" type="Promise<void>">
  Signale à Auth0 une erreur de l’API WebAuthn côté client (provenant de `navigator.credentials.get()`).
  Cette méthode doit être utilisée lorsque `useSecurityKey()` (ou un appel manuel à `navigator.credentials.get()`)
  échoue en raison d’un problème WebAuthn côté navigateur (par exemple, `NotAllowedError` si l’utilisateur annule l’invite,
  `NotFoundError` si aucun authentificateur correspondant n’est trouvé, ou un délai d’expiration).
  Elle envoie les détails de l’erreur avec `action: "showError::{errorDetailsJsonString}"` et une `response` vide.

  Promesse qui se résout lorsque le rapport d’erreur est soumis.
  Auth0 peut recharger la page avec les informations sur l’erreur ou effectuer une redirection.

  #### Lève

  Lève une erreur si l’envoi du formulaire échoue (par exemple, erreur réseau, `state` invalide).
