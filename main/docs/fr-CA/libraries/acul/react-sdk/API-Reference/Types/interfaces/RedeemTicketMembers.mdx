---
title: "RedeemTicketMembers"
---

Interface qui décrit les data disponibles sur l&#39;écran « Redeem Ticket ».

````ts Example
export interface RedeemTicketMembers extends BaseMembers {
  client: ClientMembers;
  organization: OrganizationMembers;
  prompt: PromptMembers;
  screen: ScreenMembers;
  transaction: TransactionMembers;

  /**
   * Exécute l'action par défaut sur l'écran redeem-ticket, ce qui implique habituellement de poursuivre le flux.
   * @param {CustomOptions} [payload] - Options personnalisées facultatives à inclure dans la requête.
   * @returns {Promise<void>} Une promesse qui se résout une fois l'action terminée.
   * @throws {Error} Si l'opération échoue.
   * @example
   * ```typescript
   * import RedeemTicket from '@auth0/auth0-acul-js/redeem-ticket';
   *
   * const redeemTicket = new RedeemTicket();
   * await redeemTicket.continue();
   * ```
   */
  continue(payload?: CustomOptions): Promise<void>;
}
````
