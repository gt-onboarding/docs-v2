---
title: "DeviceCodeConfirmationMembers"
---

Interface qui décrit les membres de l’écran de confirmation du code d’appareil.

```ts Example
export interface DeviceCodeConfirmationMembers extends BaseMembers {
  screen: ScreenMembersOnDeviceCodeConfirmation;

  /**
   * Confirms the device code.
   * @param payload Optional custom options to include with the request.
   */
  confirm(payload?: CustomOptions): Promise<void>;

  /**
   * Annule le flux de code de l'appareil.
   * @param payload Options personnalisées facultatives à inclure dans la requête.
   */
  cancel(payload?: CustomOptions): Promise<void>;
}
```
