export interface ScreenMembersOnResetPasswordMfaWebAuthnRoamingChallenge extends ScreenMembers {
  /**
   * Un accesseur direct pour les données `passkey.public_key` provenant de `screen.data`.
   * Cet objet contient le défi et les autres options nécessaires pour l'appel
   * à l'API WebAuthn `navigator.credentials.get()`.
   * Il est `null` si `screen.data.passkey.public_key` n'est pas disponible.
   * @type {PasskeyRead['public_key'] | null}
   * @public
   */
  publicKey: PasskeyRead['public_key'] | null;

  /**
   * Un accesseur direct pour l'indicateur `show_remember_device` provenant de `screen.data`.
   * Indique si l'interface utilisateur doit présenter une option à l'utilisateur pour mémoriser cet appareil ou navigateur
   * afin de potentiellement contourner l'AMF (MFA) lors des connexions futures. La valeur par défaut est `false` si elle n'est pas définie dans `screen.data`.
   * @type {boolean}
   * @public
   */
  showRememberDevice: boolean;
}
