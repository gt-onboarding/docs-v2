```php router.php lines 
<?php

  declare(strict_types=1);

  use Steampixel\Route;

  if ($_SERVER['HTTP_HOST'] !== 'localhost:3000') {
    die('<p>このクイックスタートは <a href="http://localhost:3000">http://localhost:3000</a> から実行するように設定されています。</p>');
  }

  Route::add('/', function() use ($sdk) {
    require('profile.php');
  });

  Route::add('/login', function() use ($sdk) {
    require('login.php');
  });

  Route::add('/callback', function() use ($sdk) {
    require('callback.php');
  });

  Route::add('/logout', function() use ($sdk) {
    require('logout.php');
  });

  Route::run();
```
