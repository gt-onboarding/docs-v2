```typescript src/app/page.tsx lines 
import { auth0 } from "@/lib/auth0";
import './globals.css';

export default async function Home() {
  // ユーザーセッションを取得
  const session = await auth0.getSession();

  // セッションがない場合、サインアップとログインボタンを表示
  if (!session) {
    return (
      <main>
        <a href="/auth/login?screen_hint=signup">
          <button>サインアップ</button>
        </a>
        <a href="/auth/login">
          <button>ログイン</button>
        </a>
      </main>
    );
  }

  // セッションが存在する場合、ウェルカムメッセージとログアウトボタンを表示
  return (
    <main>
      <h1>{session.user.name}さん、ようこそ!</h1>
      <p>
        <a href="/auth/logout">
          <button>ログアウト</button>
        </a>
      </p>
    </main>
  );
}
```
