from authlib.integrations.django_oauth2 import ResourceProtector
from django.http import JsonResponse
from . import validator

require_auth = ResourceProtector()
validator = validator.Auth0JWTBearerTokenValidator(
    "{yourDomain}",
    "{yourApiIdentifier}"
)
require_auth.register_token_validator(validator)

def public(request):
    """このルートにアクセスするためにアクセストークンは不要です
    """
    response = "パブリックエンドポイントからこんにちは!これを表示するために認証する必要はありません。"
    return JsonResponse(dict(message=response))

@require_auth(None)
def private(request):
    """このルートにアクセスするには有効なアクセストークンが必要です
    """
    response = "プライベートエンドポイントからこんにちは!これを表示するには認証する必要があります。"
    return JsonResponse(dict(message=response))

@require_auth("read:messages")
def private_scoped(request):
    """このルートにアクセスするには有効なアクセストークンと適切なスコープが必要です
    """
    response = "プライベートエンドポイントからこんにちは!これを表示するには認証する必要があり、read:messagesのスコープが必要です。"
    return JsonResponse(dict(message=response))
