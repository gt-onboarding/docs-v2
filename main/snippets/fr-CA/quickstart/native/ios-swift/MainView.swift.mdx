import SwiftUI
import Auth0

struct MainView: View {
    @State var user: User?

    var body: some View {
        if let user = self.user {
            VStack {
                ProfileView(user: user)
                Button("Déconnexion", action: self.logout)
            }
        } else {
            Button("Connexion", action: self.login)
        }
    }
}

extension MainView {
    func login() {
        Auth0
            .webAuth()
            .useHTTPS() // Utilise une URL de rappel Universal Link sur iOS 17.4+ / macOS 14.4+
            .start { result in
                switch result {
                case .success(let credentials):
                    self.user = User(from: credentials.idToken)
                case .failure(let error):
                    print("Échec avec : \(error)")
                }
            }
    }

    func logout() {
        Auth0
            .webAuth()
            .useHTTPS() // Utilise une URL de déconnexion Universal Link sur iOS 17.4+ / macOS 14.4+
            .clearSession { result in
                switch result {
                case .success:
                    self.user = nil
                case .failure(let error):
                    print("Échec avec : \(error)")
                }
            }
    }
}
