import { Component, OnInit } from '@angular/core';
import { AuthService } from '@auth0/auth0-angular';
import { Browser } from '@capacitor/browser';
import { tap } from 'rxjs/operators';

// Construire l'URL de retour vers votre application après la déconnexion
const returnTo = `${config.appId}://{yourDomain}/capacitor/${config.appId}/callback`;

@Component({
  selector: 'app-logout-button',
  template: `<ion-button (click)="logout()">Se déconnecter</ion-button>`,
})
export class LogoutButtonComponent {
  // Importer le module AuthService de la trousse de développement logiciel (SDK) Angular Auth0
  constructor(public auth: AuthService) {}

   logout() {
    this.auth
      .logout({ 
        logoutParams: {
          returnTo,
        },
        async openUrl(url: string) {
          await Browser.open({ url });
        } 
      })
      .subscribe();
  }
}
