import { useAuth0 } from '@auth0/auth0-react';
import { Browser } from '@capacitor/browser';
import { IonButton } from '@ionic/react';

// Ceci devrait correspondre à l'URL ajoutée précédemment à votre paramètre « URL de déconnexion autorisées »
// dans le Auth0 Dashboard.
const logoutUri = 'YOUR_PACKAGE_ID://{yourDomain}/capacitor/YOUR_PACKAGE_ID/callback';

const LogoutButton: React.FC = () => {
  const { logout } = useAuth0();

  const doLogout = async () => {
    await logout({
      logoutParams: {
        returnTo: logoutUri
      },
      async openUrl(url) {
         // Redirection à l'aide du plugin Browser de Capacitor
        await Browser.open({
          url,
          windowName: "_self"
        });
      }
    });
  };

  return <IonButton onClick={doLogout}>Se déconnecter</IonButton>;
};

export default LogoutButton;
