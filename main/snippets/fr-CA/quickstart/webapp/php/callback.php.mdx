<?php

  declare(strict_types=1);

  /**
   * Au retour de Universal Login d'Auth0, nous devons effectuer un échange de code en utilisant la méthode `exchange()`
   * pour terminer le flux d'authentification. Ce processus configure la session pour utilisation par l'application.
   *
   * En cas de succès, l'utilisateur sera redirigé vers la route index.
   */

   $hasAuthenticated = isset($_GET['state']) && isset($_GET['code']);
   $hasAuthenticationFailure = isset($_GET['error']);

  // L'utilisateur final sera renvoyé avec les valeurs ?state et ?code dans sa requête en cas de succès.
  if ($hasAuthenticated) {
    try {
      $sdk->exchange();
    } catch (\Throwable $th) {
      printf('Impossible de terminer l\'authentification : %s', $th->getMessage());
      exit;
    }
  }

  // Lorsque l'authentification échoue, l'utilisateur final sera renvoyé avec une ?error dans sa requête.
  if ($hasAuthenticationFailure) {
    printf('Échec de l\'authentification : %s', htmlspecialchars(strip_tags(filter_input(INPUT_GET, 'error'))));
    exit;
  }

  // Rien à faire : rediriger vers la route index.
  header('Location: /');
