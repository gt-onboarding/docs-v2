"""Exemple d'intégration d'Auth0 avec l'API Python Flask
    """

    from os import environ as env

    from dotenv import load_dotenv, find_dotenv
    from flask import Flask, jsonify
    from authlib.integrations.flask_oauth2 import ResourceProtector
    from validator import Auth0JWTBearerTokenValidator

    require_auth = ResourceProtector()
    validator = Auth0JWTBearerTokenValidator(
"{yourDomain}",
"{yourApiIdentifier}"
    )
    require_auth.register_token_validator(validator)

    APP = Flask(__name__)

    @APP.route("/api/public")
    def public():
"""Aucun jeton d'accès n'est requis."""
response = (
    "Bonjour d'un point de terminaison public! Vous n'avez pas besoin d'être"
    " authentifié pour voir ceci."
)
return jsonify(message=response)

    @APP.route("/api/private")
    @require_auth(None)
    def private():
"""Un jeton d'accès valide est requis."""
response = (
    "Bonjour d'un point de terminaison privé! Vous devez être"
    " authentifié pour voir ceci."
)
return jsonify(message=response)

    @APP.route("/api/private-scoped")
    @require_auth("read:messages")
    def private_scoped():
"""Un jeton d'accès valide et une portée sont requis."""
response = (
    "Bonjour d'un point de terminaison privé! Vous devez être"
    " authentifié et avoir une portée de read:messages pour voir"
    " ceci."
)
return jsonify(message=response)
