from authlib.integrations.django_oauth2 import ResourceProtector
from django.http import JsonResponse
from . import validator

require_auth = ResourceProtector()
validator = validator.Auth0JWTBearerTokenValidator(
    "{yourDomain}",
    "{yourApiIdentifier}"
)
require_auth.register_token_validator(validator)

def public(request):
    """Aucun jeton d'accès n'est requis pour accéder à cette route
    """
    response = "Bonjour d'un point de terminaison public! Vous n'avez pas besoin d'être authentifié pour voir ceci."
    return JsonResponse(dict(message=response))

@require_auth(None)
def private(request):
    """Un jeton d'accès valide est requis pour accéder à cette route
    """
    response = "Bonjour d'un point de terminaison privé! Vous devez être authentifié pour voir ceci."
    return JsonResponse(dict(message=response))

@require_auth("read:messages")
def private_scoped(request):
    """Un jeton d'accès valide et une portée appropriée sont requis pour accéder à cette route
    """
    response = "Bonjour d'un point de terminaison privé! Vous devez être authentifié et avoir une portée de read:messages pour voir ceci."
    return JsonResponse(dict(message=response))
